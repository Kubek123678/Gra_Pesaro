<script>
  let countdownTimer;

  function showSection(id) {
    document.querySelectorAll('section').forEach(s => s.classList.remove('active'));
    document.getElementById(id).classList.add('active');
  }
  function registerTeam() {
    const name = document.getElementById('teamName').value;
    localStorage.setItem('teamName', name);
    alert('Zarejestrowano drużynę: ' + name);
  }
  function startGame() {
    const hours = parseInt(document.getElementById('adminTime').value);
    const end = Date.now() + hours * 3600000;
    localStorage.setItem('gameEnd', end);
    updateCountdown();
    countdownTimer = setInterval(updateCountdown, 1000);
    alert('Gra rozpoczęta!');
  }
  function updateCountdown() {
    const end = localStorage.getItem('gameEnd');
    if (!end) return;
    const now = Date.now();
    const diff = end - now;
    if (diff <= 0) {
      document.getElementById('countdown').innerText = 'Gra zakończona';
      clearInterval(countdownTimer);
      return;
    }
    const h = Math.floor(diff / 3600000);
    const m = Math.floor((diff % 3600000) / 60000);
    const s = Math.floor((diff % 60000) / 1000);
    document.getElementById('countdown').innerText = `Czas: ${h}h ${m}m ${s}s`;
  }
  function uploadImage() {
    alert('Zdjęcie przesłane (symulacja)');
  }
  function loginAdmin() {
    const pass = document.getElementById('adminPass').value;
    if (pass === 'pesaro2025') {
      document.getElementById('adminPanel').style.display = 'block';
      const select = document.getElementById('adminTeam');
      select.innerHTML = '';
      const teams = ['Barcelona', 'Kotki'];
      teams.forEach(t => {
        const option = document.createElement('option');
        option.value = t;
        option.text = t;
        select.appendChild(option);
      });
    } else alert('Błędne hasło');
  }
  function addPoints() {
    const team = document.getElementById('adminTeam').value;
    const pts = parseInt(document.getElementById('adminPoints').value);
    const scores = JSON.parse(localStorage.getItem('scores') || '{}');
    scores[team] = (scores[team] || 0) + pts;
    localStorage.setItem('scores', JSON.stringify(scores));
    alert('Punkty dodane');
    renderRanking();
  }
  function renderRanking() {
    const scores = JSON.parse(localStorage.getItem('scores') || '{}');
    let html = '';
    Object.entries(scores).sort((a, b) => b[1] - a[1]).forEach(([team, pts]) => {
      html += `<p><strong>${team}</strong>: ${pts} pkt</p>`;
    });
    document.getElementById('rankingList').innerHTML = html;
  }
  function sendAdminMessage() {
    const msg = document.getElementById('adminMsg').value;
    const chat = JSON.parse(localStorage.getItem('chat') || '[]');
    chat.push({ user: 'Admin', text: msg });
    localStorage.setItem('chat', JSON.stringify(chat));
    document.getElementById('adminMsg').value = '';
    renderChat();
  }
  function sendChat() {
    const user = localStorage.getItem('teamName') || 'Nieznana';
    const text = document.getElementById('chatInput').value;
    const chat = JSON.parse(localStorage.getItem('chat') || '[]');
    chat.push({ user, text });
    localStorage.setItem('chat', JSON.stringify(chat));
    document.getElementById('chatInput').value = '';
    renderChat();
  }
  function renderChat() {
    const chat = JSON.parse(localStorage.getItem('chat') || '[]');
    const box = document.getElementById('chatBox');
    box.innerHTML = chat.map(c => `<p><strong>${c.user}</strong>: ${c.text}</p>`).join('');
    box.scrollTop = box.scrollHeight;
  }
  setInterval(renderChat, 3000);
  renderRanking();
  updateCountdown();
</script>
